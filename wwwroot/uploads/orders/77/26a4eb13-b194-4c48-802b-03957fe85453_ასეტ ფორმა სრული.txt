          <Section title="ძირითადი ინფორმაცია">

            <Field name="manufacturer" label="მწარმოებელი" form={form} />

            <Field name="barcode" label="ბარკოდი" form={form} />
            <Field name="serialNumber" label="სერიული ნომერი" form={form} />
            <Field name="inventoryNumber" label="ინვენტარნი ნომერი" form={form} />
            <WideField name="description" label="აღწერა" form={form} />
            <WideField name="searchDescription" label="ძებნის აღწერა" form={form} />
          </Section>

          {/* კლასიფიკაცია */}
          <Section title="კლასიფიკაცია">
            <SelectField name="categoryId" label="კატეგორია" items={categories} loading={catLoading} form={form} />
            <SelectField name="departmentId" label="დეპარტამენტი" items={departments} loading={depLoading} form={form} />

            <SelectField name="assetStatusId" label="სტატუსი" items={statuses} loading={statusLoading} form={form} />
            <SelectField name="responsiblePersonId" label="პასუხისმგებელი პირი" items={employees} loading={empLoading} form={form} />
            <SelectField name="supplierId" label="მომწოდებელი" items={suppliers} loading={supLoading} form={form} />
            <SelectField name="parentAssetId" label="მშობელი აქტივი" items={[]} loading={false} form={form} placeholder="მალე..." disabled />

          <Section title="ფინანსური ინფორმაცია">

            <Field name="usefulLifeMonths" label="სარგებლობის ვადა (თვე)" type="number" form={form} />
            <Field name="currency" label="ვალუტა" form={form} />
            <Field name="disposalValue" label="ჩამოწერის ღირებულება" type="number" form={form} />

            </Button>

        </form>
      </Form>
    </DialogContent>
  )
}

/* ================= Helpers (უცვლელი) ================= */
// Section, Field, WideField, SelectField — შენი წინა კოდიდან ჩასვი აქ
/* ================= Helper Components ================= */
function Section({ title, children }: { title: string; children: React.ReactNode }) {
  return (
    <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-x-6 gap-y-4 border-b pb-6 last:border-0">
      <h3 className="col-span-full text-lg font-semibold">{title}</h3>
      {children}
    </div>
  )
}

function Field({
  name,
  label,
  type = "text",
  form,
}: {
  name: keyof AssetFormValues
  label: string
  type?: string
  form: any
}) {
  return (
    <FormField
      control={form.control}
      name={name}
      render={({ field }) => (
        <FormItem>
          <FormLabel>{label}</FormLabel>
          <FormControl>
            <Input
              type={type}
              {...field}
              value={field.value ?? ""}
              onChange={(e) => {
                const v = e.target.value
                field.onChange(type === "number" ? (v === "" ? null : Number(v)) : v)
              }}
            />
          </FormControl>
          <FormMessage />
        </FormItem>
      )}
    />
  )
}

function WideField({
  name,
  label,
  form,
}: {
  name: keyof AssetFormValues
  label: string
  form: any
}) {
  return (
    <FormField
      control={form.control}
      name={name}
      render={({ field }) => (
        <FormItem className="col-span-full">
          <FormLabel>{label}</FormLabel>
          <FormControl>
            <Textarea
              rows={3}
              {...field}
              value={field.value ?? ""}
            />
          </FormControl>
          <FormMessage />
        </FormItem>
      )}
    />
  )
}

function SelectField({
  name,
  label,
  items,
  loading,
  form,
  placeholder = "აირჩიეთ...",
  disabled = false,
}: {
  name: keyof AssetFormValues
  label: string
  items: LookupItem[]
  loading: boolean
  form: any
  placeholder?: string
  disabled?: boolean
}) {
  return (
    <FormField
      control={form.control}
      name={name}
      render={({ field }) => (
        <FormItem>
          <FormLabel>{label}</FormLabel>
          <Select
            onValueChange={(value) => field.onChange(value ? Number(value) : null)}
            value={field.value?.toString() || ""}
            disabled={loading || disabled}
          >
            <FormControl>
              <SelectTrigger>
                <SelectValue placeholder={loading ? "იტვირთება..." : placeholder} />
              </SelectTrigger>
            </FormControl>
            <SelectContent>
              {items.map((item) => (
                <SelectItem key={item.id} value={item.id.toString()}>
                  {item.name}
                </SelectItem>
              ))}
            </SelectContent>
          </Select>
          <FormMessage />
        </FormItem>
      )}
    />
  )
}
